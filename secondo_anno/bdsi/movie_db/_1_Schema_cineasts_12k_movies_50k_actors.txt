CREATE TABLE ATTORE (
    CODATTORE NUMBER PRIMARY KEY,
    NOME VARCHAR2(50),
    COGNOME VARCHAR2(50),
    NAZIONALITA VARCHAR2(50)
);

CREATE TABLE FILM (
    CODFILM NUMBER PRIMARY KEY,
    TITOLO VARCHAR2(100),
    GENERE VARCHAR2(50),
    DURATA NUMBER,
    ANNO NUMBER,
    INCASSO NUMBER,
    VOTO NUMBER
);

CREATE TABLE INTERPRETA (
    CODFILM NUMBER,
    CODATTORE NUMBER,
    RUOLO VARCHAR2(100),
    COMPENSO NUMBER,
    PRIMARY KEY (CODFILM, CODATTORE),
    CONSTRAINT FK_INTERPRETA_FILM FOREIGN KEY (CODFILM)
        REFERENCES FILM (CODFILM),
    CONSTRAINT FK_INTERPRETA_ATTORE FOREIGN KEY (CODATTORE)
        REFERENCES ATTORE (CODATTORE)
);
